def match(text)
  if m = @regexp.match(text)
    format = ''
    format << "%a, " if m['weekday']
    format << "%d %b %Y"
    format << " %H:%M" if m['time']
    format << ":%S" if m['second']
    format << " %z" if m['zone_off']
    format << " %Z" if m['zone_abb']
    SimpleMatch.new(format)
  else
    nil
  end
end
